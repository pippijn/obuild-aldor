\section{Character codes}

\adtype{AnsiCodes} and
\adtype{CharacterNames} 
constitute the character code part of \projectname.

\adtype{AnsiCodes} provides character codes to color text and move the cursor.

\inputSourceFile{ansicodes/redonblack.as}{An example of using \adtype{AnsiCodes}}
%\inputOutput{ansicodes/redonblack}

\adtype{CharacterNames} provides Unicode \cite{Unicode} character names and aliases for $7$-bit ASCII codes.
\inputSourceFile{unicode/unicode.as}{An example of using \adtype{CharacterNames}}
\pagebreak
\inputOutput{unicode/unicode}
